#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main() {
	// your code goes here
	vector <int> arr(6, 0);
	vector <int> ps(7, 0);
	int maxn=0;
	
	//taking in array
	int i=0;
	for (i=0; i<=5; i++){
	cin >>arr[i];
	}
	for (i=0; i<=5; i++){
	cout << arr[i] << " ";
	}
	
	//printing array
	printf ("\n");
	ps[0]=arr[0];
	for (i=1; i<=5; i++){
		ps[i] = ps[i-1] + arr[i];
	}
	for (i=0; i<=5; i++){
		cout <<ps[i] <<" ";
	}
	
	//max in array
	printf ("\n");
	for (i=0; i<=5; i++){
		if (arr[i] >= maxn){
			maxn = arr[i];
		}
		
	}
	cout <<maxn;
	
	//print M1.1 V3
	printf ("\n");
	int p, q, r, j, k, ans=0;
	cin >>p >>q >>r;
	for (i=0; i<=5; i++){
		for (j=i+1; j<=5; j++){
			for (k=j+1; k<=5; k++){
			ans=max(ans, p*arr[i]+q*arr[j]+r*arr[k]);
			}
		}
	}
	printf ("%d \n", ans);
	return 0;
}
